5Ô∏è‚É£ Advanced Printing

import textwrap

wrapped_string = textwrap.fill(long_string, width=50)

import time
for i in range(0, 101, 5):
    print(f"\rProgress: [{'#' * i:<100}] {i}%", end="")
    time.sleep(0.02)
